function Meta({ additional }) {
  return (
    <StaticQuery
      graphql={query}
      render={data => (
        <Helmet
          title={data.metaYaml.title}
          meta={
            [
              { name: 'description', content: data.metaYaml.description }
              { name: 'twitter:creator', content: data.metaYaml.twitter }
              // you get the idea
            ].concat(additional)
          }
        />
      )}
    />
  )
}
